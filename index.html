        function filterEncyclopedia() {

            const query = document.getElementById('search-encyclopedia').value.toLowerCase();

            const cards = document.querySelectorAll('.encyclopedia-card');

            const sections = document.querySelectorAll('.encyclopedia-section');


            cards.forEach(card => {

                const content = card.getAttribute('data-search');

                card.style.display = content.includes(query) ? 'block' : 'none';

            });


            // セクション内に表示されるカードが1つもない場合はセクションごと隠す

            sections.forEach(section => {

                const visibleCards = section.querySelectorAll('.encyclopedia-card[style="display: block;"]');

                section.style.display = visibleCards.length > 0 ? 'block' : 'none';

            });

        }


        function startDirectTraining() {

            isDirectMode = true;

            sessionQuestions = [...questions].sort(() => Math.random() - 0.5);

            currentCaseIdx = 0;

            currentStep = 1;

            showScreen('training');

            renderStep();

        }


        function startTraining(category) {

            isDirectMode = false;

            sessionQuestions = category === 'all' 

                ? [...questions].sort(() => Math.random() - 0.5)

                : questions.filter(q => q.cat === category).sort(() => Math.random() - 0.5);

            

            currentCaseIdx = 0;

            currentStep = 1;

            showScreen('training');

            renderStep();

        }


        function renderStep() {

            if (sessionQuestions.length === 0) return showScreen('home');

            

            const data = sessionQuestions[currentCaseIdx];

            document.getElementById('progress-text').innerText = `Q ${currentCaseIdx + 1} / ${sessionQuestions.length}`;

            document.getElementById('category-badge').innerText = isDirectMode ? "薬×副作用" : data.cat;

            

            document.getElementById('case-container').style.display = isDirectMode ? 'none' : 'block';

            document.getElementById('step-bars').style.display = isDirectMode ? 'none' : 'flex';


            const useClassAsMain = isDirectMode && Math.random() > 0.5;

            if (useClassAsMain) {

                document.getElementById('q-header-label').innerText = "Drug Class (薬効群名)";

                document.getElementById('drug-name').innerText = getDrugClass(data.drug);

                document.getElementById('drug-class').innerText = data.drug;

            } else {

                document.getElementById('q-header-label').innerText = "Drug Name / Class";

                document.getElementById('drug-name').innerText = data.drug;

                document.getElementById('drug-class').innerText = getDrugClass(data.drug);

            }

            

            if(!isDirectMode) {

                document.getElementById('case-text').innerText = data.case;

                document.getElementById('step-1-bar').className = `step-indicator ${currentStep === 1 ? 'step-active' : 'step-inactive'}`;

                document.getElementById('step-2-bar').className = `step-indicator ${currentStep === 2 ? 'step-active' : 'step-inactive'}`;

            }


            const container = document.getElementById('options-container');

            container.innerHTML = '';

            document.getElementById('feedback').classList.add('hidden');

            document.getElementById('next-btn').classList.add('hidden');


            const isStep1 = (currentStep === 1);

            const correct = isStep1 ? data.side : data.obs;

            const prompt = isStep1 ? "代表的な副作用を選んでください" : "看護師の重点観察項目は？";

            

            document.getElementById('question-prompt').innerText = prompt;


            // ダミー選択肢の生成

            const allOptions = questions.map(q => isStep1 ? q.side : q.obs).filter(v => v !== correct);

            const uniqueOptions = [...new Set(allOptions)];

            const fakes = shuffle(uniqueOptions).slice(0, 3);

            const opts = shuffle([correct, ...fakes]);


            opts.forEach(opt => {

                const btn = document.createElement('button');

                btn.className = "option-btn w-full bg-white p-4 rounded-2xl border-2 border-slate-100 text-left font-bold text-slate-700 text-sm leading-tight";

                btn.innerText = opt;

                btn.onclick = () => {

                    const btns = container.children;

                    for(let b of btns) b.disabled = true;

                    if(opt === correct) {

                        document.getElementById('feedback').innerText = "⭕ 正解";

                        document.getElementById('feedback').className = "p-4 rounded-2xl bg-green-50 text-green-700 font-bold border border-green-200";

                        btn.classList.add("border-green-500", "bg-green-50");

                    } else {

                        document.getElementById('feedback').innerText = `❌ 正解：${correct}`;

                        document.getElementById('feedback').className = "p-4 rounded-2xl bg-red-50 text-red-700 font-bold border border-red-200";

                        btn.classList.add("border-red-500", "bg-red-50");

                    }

                    document.getElementById('feedback').classList.remove('hidden');

                    document.getElementById('next-btn').classList.remove('hidden');

                };

                container.appendChild(btn);

            });

        }


        function processNext() {

            if(!isDirectMode && currentStep === 1) {

                currentStep = 2;

                renderStep();

            } else {

                currentStep = 1;

                currentCaseIdx++;

                if(currentCaseIdx < sessionQuestions.length) renderStep();

                else showScreen('result');

            }

        }


        function showScreen(id) {

            ['home', 'training', 'result', 'encyclopedia'].forEach(s => {

                const el = document.getElementById(s+'-screen');

                if(el) el.classList.add('hidden');

            });

            document.getElementById(id+'-screen').classList.remove('hidden');

            window.scrollTo(0, 0);

        }


        function shuffle(arr) { return [...arr].sort(() => Math.random() - 0.5); }


        function exportToCSV() {

            let csv = "分野,症例,薬剤名,副作用,観察項目\n";

            questions.forEach(q => {

                csv += `"${q.cat}","${q.case}","${q.drug}","${q.side}","${q.obs}"\n`;

            });

            const blob = new Blob(["\ufeff" + csv], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            link.href = URL.createObjectURL(blob);

            link.download = `nursing_pharma_100.csv`;

            link.click();

        }


        function importFromCSV(e) {

            const file = e.target.files[0];

            if (!file) return;

            const reader = new FileReader();

            reader.onload = function(event) {

                const text = event.target.result;

                const rows = text.split('\n').slice(1);

                const newQuestions = rows.map(row => {

                    const cols = row.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/).map(v => v.replace(/^"|"$/g, '').trim());

                    if(cols.length >= 5) {

                        return { cat: cols[0], case: cols[1], drug: cols[2], side: cols[3], obs: cols[4] };

                    }

                    return null;

                }).filter(q => q && q.cat);


                if(newQuestions.length > 0) {

                    questions = newQuestions;

                    localStorage.setItem('nursing_pharma_data', JSON.stringify(questions));

                    renderHome();

                    alert(`${newQuestions.length}件の問題を取り込みました。`);

                }

            };

            reader.readAsText(file);

        }

    </script>
